/**
 * Blocks have the following dimensions:
 *  x length: 3
 *  y length: 7
 *  z length: 1
 *
 * A block's position and rotation is calculated at
 * its center: (1.5, 3.5, 0.5)
 *
 * Render blocks by adding them to the
 * DBLOCKS.blocks array.
 *
 * Have fun playing with dblocks!
 */
(function() {
    var buildLevel = function(i, o) {
        var yrot = (i % 2 === 0) ?  Math.PI / 2 : 0;

        for (var j = 1; j <= 3; j++) {
            var xpos = (yrot !== 0) ? (j + .4) * 3 : 7;
            var zpos = (xpos === 7) ? j * 3 : 6;

            xpos += o;
            zpos += o;

            var block = {
                pos: {x: xpos, y: i * 3, z: zpos},
                rot: {x: 0, y: yrot, z: Math.PI / 2}
            };

            DBLOCKS.blocks.push(block);
        }
    };

    var buildTower = function(o) {
        for (var i = 1; i <= 12; i++) {
            buildLevel(i, o);
        }
    };

    buildTower(0);
    buildTower(8);
    buildTower(16);
})();
